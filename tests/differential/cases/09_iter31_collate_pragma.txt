# ORDER BY ... NULLS FIRST / LAST postfix (Iter31.N):
#  * SQL standard syntax sqlite3 supports since 3.30. Direction-default
#    when omitted: ASC → NULLS FIRST, DESC → NULLS LAST. Explicit NULLS
#    placement overrides the default and is independent of ASC/DESC.
#  * `nulls` / `first` / `last` are NOT reserved keywords in sqlite3 —
#    `SELECT first FROM (SELECT 1 AS first)` works; the parser must
#    detect them as identifiers in postfix position only.
SELECT * FROM (VALUES(NULL),(2),(1)) ORDER BY column1
SELECT * FROM (VALUES(NULL),(2),(1)) ORDER BY column1 DESC
SELECT * FROM (VALUES(NULL),(2),(1)) ORDER BY column1 ASC NULLS LAST
SELECT * FROM (VALUES(NULL),(2),(1)) ORDER BY column1 ASC NULLS FIRST
SELECT * FROM (VALUES(NULL),(2),(1)) ORDER BY column1 DESC NULLS LAST
SELECT * FROM (VALUES(NULL),(2),(1)) ORDER BY column1 DESC NULLS FIRST
SELECT * FROM (VALUES(NULL),(2),(1)) ORDER BY column1 NULLS LAST
SELECT * FROM (VALUES(NULL),(2),(1)) ORDER BY column1 NULLS FIRST
SELECT * FROM (VALUES(NULL),(NULL),(1),(2)) ORDER BY column1 NULLS LAST
SELECT * FROM (VALUES(1)) ORDER BY column1 NULLS first
SELECT * FROM (VALUES(1)) ORDER BY column1 nulls LAST
SELECT * FROM (VALUES(1)) ORDER BY column1 NULLS
SELECT * FROM (VALUES(1)) ORDER BY column1 NULLS bogus
SELECT first FROM (SELECT 1 AS first)
SELECT nulls FROM (SELECT 1 AS nulls)
SELECT last FROM (SELECT 1 AS last)
SELECT 1 ORDER BY 1 ASC NULLS FIRST
SELECT * FROM (VALUES(1,'a'),(2,NULL),(NULL,'c')) ORDER BY 2 NULLS FIRST, 1
SELECT * FROM (VALUES(1,'a'),(2,NULL),(NULL,'c')) ORDER BY 2 NULLS LAST, 1

# COLLATE NOCASE/BINARY/RTRIM postfix (Iter31.O):
#  * SQL standard postfix; sqlite3 ships three built-ins. Comparison-
#    time hint that wraps the operand: `peekKind` extracts the outermost
#    wrapper per side, "left wins" picks LHS over RHS over BINARY.
#  * COLLATE binds tighter than `<`/`=`/etc — RHS of binary comparison
#    catches it via parseUnary→parseCollatePostfix. Operators handled
#    inside parseEquality (IN/IS/BETWEEN) miss the right operand path,
#    so parseExpr has a trailing-COLLATE rescue (no-op on non-TEXT).
#  * COLLATE is a fully reserved keyword — `SELECT collate` errors.
#  * `BINARY` / `NOCASE` / `RTRIM` are case-insensitive identifier names.
#  * Sort is stable: equal-under-collation rows preserve input order
#    (verified `('A','a','B','b') ORDER BY col COLLATE NOCASE` returns
#    input verbatim — switched select_post.sortRowsByKeys from pdq to
#    block sort to honor this).
SELECT 'A' COLLATE NOCASE = 'a'
SELECT 'A' COLLATE BINARY = 'a'
SELECT 'A' = 'a' COLLATE NOCASE
SELECT 'a' COLLATE NOCASE = 'A' COLLATE BINARY
SELECT 'A' COLLATE NOCASE = 'a' COLLATE BINARY
SELECT 'A' COLLATE noCASE = 'a'
SELECT 'a   ' COLLATE RTRIM = 'a'
SELECT 'a' COLLATE RTRIM = 'a   '
SELECT 'a  b' COLLATE RTRIM = 'a  b   '
SELECT 1 COLLATE NOCASE = 1
SELECT typeof('A' COLLATE NOCASE)
SELECT NULL COLLATE NOCASE = 'a'
SELECT NULL COLLATE NOCASE IS NULL
SELECT 'A' COLLATE NOCASE IS 'a'
SELECT 'A' COLLATE NOCASE IS NOT 'a'
SELECT 'A' COLLATE NOCASE IS DISTINCT FROM 'a'
SELECT 'A' COLLATE NOCASE BETWEEN 'a' AND 'z'
SELECT 'A' BETWEEN 'a' COLLATE NOCASE AND 'z' COLLATE NOCASE
SELECT 'A' COLLATE NOCASE IN ('a', 'b', 'c')
SELECT 'A' IN ('a', 'b', 'c') COLLATE NOCASE
SELECT 'A' COLLATE NOCASE NOT IN ('x', 'y')
SELECT 'a' < 'B' COLLATE NOCASE
SELECT -5 COLLATE NOCASE = -5
SELECT abs(-5) COLLATE NOCASE = 5
SELECT length('A' COLLATE NOCASE)
SELECT 'A' COLLATE BOGUS = 'a'
SELECT 'A' COLLATE = 'a'
SELECT collate FROM (SELECT 1 AS collate)
SELECT * FROM (VALUES('A'),('b'),('a'),('B')) ORDER BY column1 COLLATE NOCASE
SELECT * FROM (VALUES('A'),('b'),('a'),('B')) ORDER BY column1 COLLATE NOCASE DESC
SELECT * FROM (VALUES('a',1),('A',2),('b',3),('B',4)) ORDER BY column1 COLLATE NOCASE, column2
SELECT * FROM (VALUES('A'),('a'),('B'),('b')) ORDER BY 1 COLLATE NOCASE
SELECT * FROM (VALUES('a   '),('a'),('a  ')) ORDER BY column1 COLLATE RTRIM
SELECT * FROM (VALUES(NULL),('A'),('a'),(1),(NULL)) ORDER BY column1 COLLATE NOCASE
SELECT (('a' COLLATE BINARY) COLLATE NOCASE) = 'A'
SELECT 'a' COLLATE NOCASE COLLATE BINARY = 'A'

# GROUP BY ... COLLATE NOCASE/BINARY/RTRIM (Iter31.P):
#  * GROUP BY honors per-term COLLATE clause for key equality.
#  * Default (no COLLATE) is BINARY (bytewise) — case differs ⇒ split groups.
#  * NOCASE folds 'A'/'a' into one group; representative is the first row.
#  * RTRIM folds 'a'/'a   '/'a  ' into one group.
#  * Literal 0 in GROUP BY is NOT a position reference (sqlite3 quirk; only
#    positive ints fold to position).
#  * Mixed COLLATE per-term: each column key compares under its own kind.
#  * NULL groups still collapse to one (NULL == NULL inside group key).
SELECT x FROM (SELECT 'A' AS x UNION ALL SELECT 'a') GROUP BY x ORDER BY x
SELECT x FROM (SELECT 'A' AS x UNION ALL SELECT 'a') GROUP BY x COLLATE NOCASE
SELECT x FROM (SELECT 'A' AS x UNION ALL SELECT 'a') GROUP BY x COLLATE BINARY ORDER BY x
SELECT x, count(*) FROM (SELECT 'A' AS x UNION ALL SELECT 'a' UNION ALL SELECT 'B') GROUP BY x COLLATE NOCASE ORDER BY x COLLATE NOCASE
SELECT count(*) FROM (SELECT 'A' AS x UNION ALL SELECT 'a' UNION ALL SELECT 'B' UNION ALL SELECT 'b') GROUP BY x COLLATE NOCASE ORDER BY x COLLATE NOCASE
SELECT x, count(*) FROM (SELECT 'a' AS x UNION ALL SELECT 'a   ' UNION ALL SELECT 'a  ') GROUP BY x COLLATE RTRIM
SELECT x, count(*) FROM (SELECT 'a' AS x UNION ALL SELECT 'a   ' UNION ALL SELECT 'a  ') GROUP BY x COLLATE BINARY ORDER BY x
SELECT x, count(*) FROM (SELECT 'A' AS x UNION ALL SELECT 'a' UNION ALL SELECT NULL UNION ALL SELECT NULL) GROUP BY x COLLATE NOCASE ORDER BY x COLLATE NOCASE
SELECT x, sum(y) FROM (SELECT 'A' AS x, 1 AS y UNION ALL SELECT 'a', 2 UNION ALL SELECT 'B', 3) GROUP BY x COLLATE NOCASE ORDER BY x COLLATE NOCASE
SELECT x, count(*) FROM (SELECT 'A' AS x UNION ALL SELECT 'a' UNION ALL SELECT 'B') GROUP BY x COLLATE NOCASE HAVING count(*) > 1 ORDER BY x COLLATE NOCASE
SELECT x, y, count(*) FROM (SELECT 'A' AS x, 'X' AS y UNION ALL SELECT 'a', 'x' UNION ALL SELECT 'A', 'X') GROUP BY x, y COLLATE NOCASE ORDER BY x, y
SELECT x, y, count(*) FROM (SELECT 'A' AS x, 'X' AS y UNION ALL SELECT 'a', 'x' UNION ALL SELECT 'A', 'X') GROUP BY x COLLATE NOCASE, y COLLATE NOCASE ORDER BY x COLLATE NOCASE, y COLLATE NOCASE
SELECT name AS x, count(*) FROM (SELECT 'A' AS name UNION ALL SELECT 'a') GROUP BY x COLLATE NOCASE
SELECT name, count(*) FROM (SELECT 'A' AS name UNION ALL SELECT 'a' UNION ALL SELECT 'B') GROUP BY 1 COLLATE NOCASE ORDER BY 1 COLLATE NOCASE
SELECT name, count(*) FROM (SELECT 'A' AS name UNION ALL SELECT 'a') GROUP BY name COLLATE NOCASE

# DISTINCT ... COLLATE NOCASE/BINARY/RTRIM (Iter31.Q):
#  * SELECT DISTINCT x COLLATE NOCASE folds 'A'/'a' into one row (first wins).
#  * count(DISTINCT x COLLATE NOCASE) skips NULL and folds case-equivalent.
#  * sum/group_concat (DISTINCT x COLLATE) honor the wrapper for dedup.
#  * Multi-column DISTINCT keeps per-column collation independent.
#  * NULL still survives DISTINCT (one extra row vs case-folded-only).
#  * RTRIM folds trailing-space variants under DISTINCT.
SELECT DISTINCT x COLLATE NOCASE FROM (SELECT 'A' AS x UNION ALL SELECT 'a' UNION ALL SELECT 'b' UNION ALL SELECT 'B') ORDER BY x COLLATE NOCASE
SELECT DISTINCT x COLLATE BINARY FROM (SELECT 'A' AS x UNION ALL SELECT 'a' UNION ALL SELECT 'b') ORDER BY x
SELECT DISTINCT x COLLATE RTRIM FROM (SELECT 'a' AS x UNION ALL SELECT 'a   ' UNION ALL SELECT 'b')
SELECT count(DISTINCT x COLLATE NOCASE) FROM (SELECT 'A' AS x UNION ALL SELECT 'a' UNION ALL SELECT 'b')
SELECT count(DISTINCT x) FROM (SELECT 'A' AS x UNION ALL SELECT 'a' UNION ALL SELECT 'b')
SELECT count(DISTINCT x COLLATE NOCASE) FROM (SELECT 'A' AS x UNION ALL SELECT 'a' UNION ALL SELECT NULL UNION ALL SELECT 'B')
SELECT sum(DISTINCT y) FROM (SELECT 'A' AS x, 1 AS y UNION ALL SELECT 'a', 2 UNION ALL SELECT 'b', 1)
SELECT group_concat(DISTINCT x COLLATE NOCASE) FROM (SELECT 'A' AS x UNION ALL SELECT 'a' UNION ALL SELECT 'B')
SELECT DISTINCT x, y FROM (SELECT 'A' AS x, 1 AS y UNION ALL SELECT 'a', 1)
SELECT DISTINCT x COLLATE NOCASE, y FROM (SELECT 'A' AS x, 1 AS y UNION ALL SELECT 'a', 1 UNION ALL SELECT 'A', 2) ORDER BY x COLLATE NOCASE, y

# Set-op (UNION/INTERSECT/EXCEPT) ... COLLATE precedence (Iter31.Q):
#  * "Leftmost branch with ANY explicit COLLATE wins" — explicit BINARY on
#    LHS locks out a NOCASE on RHS (no fold), but no wrapper on LHS yields
#    to RHS's NOCASE (fold).
#  * UNION ALL never dedups so COLLATE is observationally irrelevant.
SELECT x COLLATE NOCASE FROM (SELECT 'a' AS x UNION SELECT 'A') ORDER BY 1
SELECT x FROM (SELECT 'a' AS x UNION SELECT 'A' COLLATE NOCASE) ORDER BY 1
SELECT x FROM (SELECT 'a' AS x COLLATE BINARY UNION SELECT 'A' COLLATE NOCASE) ORDER BY 1
SELECT x FROM (SELECT 'a' AS x COLLATE NOCASE INTERSECT SELECT 'A') ORDER BY 1
SELECT x FROM (SELECT 'a' AS x COLLATE RTRIM UNION SELECT 'a   ') ORDER BY 1
SELECT x FROM (SELECT 'a' AS x UNION ALL SELECT 'A' COLLATE NOCASE) ORDER BY 1

# CREATE TABLE column-level COLLATE (Iter31.R):
#  * Schema-declared `x TEXT COLLATE NOCASE` makes bare `x = 'a'` case-insensitive.
#  * Explicit COLLATE on either operand overrides the column default
#    (BINARY locks; NOCASE confirms).
#  * LHS column collation wins in column-vs-column compares
#    (a NOCASE / b BINARY: a=b honors NOCASE, b=a honors BINARY).
#  * ORDER BY on a NOCASE-declared column folds case (stable input order).
#  * GROUP BY / SELECT DISTINCT / count(DISTINCT) on a NOCASE column fold case.
#  * Wrapped expression (upper(x), x||'') breaks column-collation propagation.
#  * Multi-COLLATE on declaration: last wins.
#  * RTRIM column folds trailing-space variants on equality / DISTINCT.
#  * Unknown collation in CREATE TABLE → SyntaxError ("no such collation").
CREATE TABLE t(x TEXT COLLATE NOCASE); INSERT INTO t VALUES ('A'),('a'),('B'),('b'); SELECT count(*) FROM t WHERE x = 'a';
CREATE TABLE t(x TEXT COLLATE NOCASE); INSERT INTO t VALUES ('A'),('a'); SELECT count(*) FROM t WHERE x = 'a' COLLATE BINARY;
CREATE TABLE t(x TEXT COLLATE NOCASE); INSERT INTO t VALUES ('A'),('a'); SELECT count(*) FROM t WHERE x COLLATE BINARY = 'a';
CREATE TABLE t(a TEXT COLLATE NOCASE, b TEXT); INSERT INTO t VALUES ('A','a'); SELECT a=b, b=a FROM t;
CREATE TABLE t(x TEXT COLLATE NOCASE); INSERT INTO t VALUES ('A'),('a'),('B'),('b'); SELECT x FROM t ORDER BY x;
CREATE TABLE t(x TEXT COLLATE NOCASE); INSERT INTO t VALUES ('A'),('a'),('B'),('b'); SELECT x FROM t ORDER BY x DESC;
CREATE TABLE t(x TEXT COLLATE NOCASE); INSERT INTO t VALUES ('A'),('a'),('B'),('b'); SELECT DISTINCT x FROM t;
CREATE TABLE t(x TEXT COLLATE NOCASE); INSERT INTO t VALUES ('A'),('a'),('B'),('b'); SELECT x, count(*) FROM t GROUP BY x ORDER BY x;
CREATE TABLE t(x TEXT COLLATE NOCASE); INSERT INTO t VALUES ('A'),('a'),('B'),('b'); SELECT count(DISTINCT x) FROM t;
CREATE TABLE t(x TEXT COLLATE NOCASE); INSERT INTO t VALUES ('A'),('a'),('B'); SELECT count(*) FROM t WHERE upper(x) = 'a';
CREATE TABLE t(x TEXT COLLATE NOCASE); INSERT INTO t VALUES ('A'),('a'); SELECT count(*) FROM t WHERE x||'' = 'a';
CREATE TABLE t(x TEXT COLLATE BINARY COLLATE NOCASE); INSERT INTO t VALUES ('A'); SELECT count(*) FROM t WHERE x = 'a';
CREATE TABLE t(x TEXT COLLATE NOCASE COLLATE BINARY); INSERT INTO t VALUES ('A'); SELECT count(*) FROM t WHERE x = 'a';
CREATE TABLE t(x TEXT COLLATE RTRIM); INSERT INTO t VALUES ('a   '),('a'),('b'); SELECT count(*) FROM t WHERE x = 'a';
CREATE TABLE t(x TEXT COLLATE RTRIM); INSERT INTO t VALUES ('a   '),('a'),('b'); SELECT DISTINCT x FROM t;
CREATE TABLE t(x TEXT COLLATE NOCASE); INSERT INTO t VALUES ('A'),('a'),('B'); SELECT count(*) FROM t WHERE x IN ('a');
CREATE TABLE t(x TEXT COLLATE NOCASE); INSERT INTO t VALUES ('a'),('B'),('b'); SELECT count(*) FROM t WHERE x BETWEEN 'a' AND 'a';
CREATE TABLE t(x TEXT COLLATE BOGUS);

# Subquery-in-FROM propagates column-level COLLATE (Iter31.R follow-up):
#  * Bare column-ref projection carries the inner column's schema collation
#    across the subquery boundary; SELECT *, alias, qualified ref behave
#    identically.
#  * Wrapped exprs (max(x), upper(x)) drop propagation → BINARY in outer.
#  * DISTINCT / GROUP BY in the inner SELECT keep propagation (projection
#    is still a bare ref); setop branches drop it (sqlite3 quirk).
CREATE TABLE t(x TEXT COLLATE NOCASE); INSERT INTO t VALUES ('A'),('a'); SELECT count(*) FROM (SELECT x FROM t) WHERE x = 'a';
CREATE TABLE t(x TEXT COLLATE NOCASE); INSERT INTO t VALUES ('A'),('a'); SELECT count(*) FROM (SELECT * FROM t) WHERE x = 'a';
CREATE TABLE t(x TEXT COLLATE NOCASE); INSERT INTO t VALUES ('A'),('a'); SELECT count(*) FROM (SELECT x AS y FROM t) WHERE y = 'a';
CREATE TABLE t(x TEXT COLLATE NOCASE); INSERT INTO t VALUES ('A'),('a'); SELECT count(*) FROM (SELECT a.x FROM t a) WHERE x = 'a';
CREATE TABLE t(x TEXT COLLATE NOCASE); INSERT INTO t VALUES ('A'),('a'); SELECT count(*) FROM (SELECT upper(x) AS y FROM t) WHERE y = 'A';
CREATE TABLE t(x TEXT COLLATE NOCASE); INSERT INTO t VALUES ('A'),('a'); SELECT count(*) FROM (SELECT DISTINCT x FROM t) WHERE x = 'a';
CREATE TABLE t(x TEXT COLLATE NOCASE); INSERT INTO t VALUES ('A'),('a'); SELECT count(*) FROM (SELECT x FROM t GROUP BY x) WHERE x = 'a';
CREATE TABLE t(x TEXT COLLATE NOCASE); INSERT INTO t VALUES ('A'),('a'); SELECT count(*) FROM (SELECT x FROM t UNION SELECT 'B') WHERE x = 'a';
CREATE TABLE t(x TEXT COLLATE NOCASE); INSERT INTO t VALUES ('A'),('a'); SELECT count(*) FROM (SELECT x FROM (SELECT x FROM t)) WHERE x = 'a';

# DISTINCT runs BEFORE ORDER BY (Iter31.S):
#  * sqlite3 dedups under DISTINCT collation FIRST (keep insertion-first
#    representative), then sorts under ORDER BY collation. sqlite0 used
#    to sort-then-dedup; the surviving representative was the post-sort
#    first one, divergent when DISTINCT and ORDER BY differ in collation.
#  * Probe: COLLATE NOCASE column + ORDER BY x COLLATE BINARY — sqlite3
#    keeps {B,a} (NOCASE classes B/b and a/A; insertion order picks B
#    and a), then BINARY-sorts to B,a. RTRIM column + ORDER BY BINARY:
#    keeps 'a   ' (first in {a,a   } RTRIM-equivalent class), then sorts.
#  * Single-collation case is unaffected (sort and dedup commute when
#    they use the same equivalence) — covered for regression-locking.
CREATE TABLE t(x TEXT COLLATE NOCASE); INSERT INTO t VALUES ('B'),('a'),('A'),('b'); SELECT x FROM (SELECT DISTINCT x FROM t ORDER BY x COLLATE BINARY);
CREATE TABLE t(x TEXT COLLATE RTRIM); INSERT INTO t VALUES ('a   '),('a'),('b'); SELECT x FROM (SELECT DISTINCT x FROM t ORDER BY x COLLATE BINARY);
CREATE TABLE t(x TEXT COLLATE NOCASE); INSERT INTO t VALUES ('B'),('a'),('A'),('b'); SELECT x FROM (SELECT DISTINCT x FROM t ORDER BY x);
CREATE TABLE t(x); INSERT INTO t VALUES ('B'),('a'),('A'),('b'); SELECT DISTINCT x COLLATE NOCASE FROM t ORDER BY x COLLATE BINARY;
SELECT DISTINCT x COLLATE NOCASE FROM (SELECT 'B' AS x UNION ALL SELECT 'a' UNION ALL SELECT 'A' UNION ALL SELECT 'b') ORDER BY x COLLATE BINARY;
CREATE TABLE t(x TEXT COLLATE NOCASE, y INT); INSERT INTO t VALUES ('B',1),('a',2),('A',3),('b',4); SELECT DISTINCT x, sum(y) FROM t GROUP BY x ORDER BY x COLLATE BINARY;

# Setop chain ORDER BY column-default propagation (Iter31.T):
#  * sqlite3 carries the leftmost branch's per-projection-item column-
#    default collation through the setop chain when ORDER BY references
#    a result column. Probe: `SELECT x FROM t UNION ALL SELECT 'C' ORDER BY x`
#    where t.x COLLATE NOCASE — sqlite3 sorts NOCASE (a,A,B,b,C);
#    sqlite0 used to sort BINARY (A,B,C,a,b) because leftmost_collations
#    was passed empty.
#  * Explicit `ORDER BY x COLLATE BINARY` overrides the column-default
#    (regression-lock).
#  * RTRIM column propagates to chain ORDER BY.
CREATE TABLE t(x TEXT COLLATE NOCASE); INSERT INTO t VALUES ('B'),('a'),('A'),('b'); SELECT x FROM t UNION ALL SELECT 'C' ORDER BY x;
CREATE TABLE t(x TEXT COLLATE NOCASE); INSERT INTO t VALUES ('B'),('a'),('A'),('b'); SELECT x FROM t UNION ALL SELECT 'C' ORDER BY x COLLATE BINARY;
CREATE TABLE t(x TEXT COLLATE RTRIM); INSERT INTO t VALUES ('a   '),('b'); SELECT x FROM t UNION ALL SELECT 'a' ORDER BY x;

# Setop DEDUP column-default propagation (Iter31.U):
#  * sqlite3 propagates the leftmost branch's column-default collation
#    into UNION/INTERSECT/EXCEPT dedup comparison, not just the explicit
#    COLLATE wrapper (Iter31.Q). Precedence per branch: explicit wrapper
#    > column-ref column-default (even BINARY) > null (literal yields).
#    The chain folds "leftmost non-null wins".
#  * UNION across NOCASE+plain: 'A' and 'a' merge (NOCASE-equal), one row
#    survives (sqlite3 dedup keep-last replaces with the right side's
#    representation).
#  * Plain leftmost + NOCASE wrapper on right: column-ref's BINARY
#    column-default locks dedup to BINARY despite RHS COLLATE NOCASE
#    (regression-lock for the "column-ref overrides RHS wrapper" rule).
#  * INTERSECT and EXCEPT exercise the same SetopKinds.merge path.
#  * Star (`*`) expansion picks up cart-resolved column-default kinds.
CREATE TABLE t1(x TEXT COLLATE NOCASE); CREATE TABLE t2(y TEXT); INSERT INTO t1 VALUES('A'),('B'); INSERT INTO t2 VALUES('a'),('b'); SELECT x FROM t1 UNION SELECT y FROM t2 ORDER BY x;
CREATE TABLE t(x TEXT); INSERT INTO t VALUES('A'),('B'); SELECT x FROM t UNION SELECT 'a' COLLATE NOCASE ORDER BY x;
CREATE TABLE t1(x TEXT COLLATE NOCASE); CREATE TABLE t2(y TEXT); INSERT INTO t1 VALUES('A'),('B'); INSERT INTO t2 VALUES('a'); SELECT x FROM t1 INTERSECT SELECT y FROM t2 ORDER BY x;
CREATE TABLE t1(x TEXT COLLATE NOCASE); CREATE TABLE t2(y TEXT); INSERT INTO t1 VALUES('A'),('B'); INSERT INTO t2 VALUES('a'); SELECT x FROM t1 EXCEPT SELECT y FROM t2 ORDER BY x;
CREATE TABLE t1(x TEXT COLLATE NOCASE); CREATE TABLE t2(x TEXT); INSERT INTO t1 VALUES('A'),('B'); INSERT INTO t2 VALUES('a'); SELECT * FROM t1 UNION SELECT * FROM t2 ORDER BY x;

# max/min collated comparison (Iter31.V):
#  * sqlite3 max(x) / min(x) honour the same collation precedence as
#    DISTINCT — explicit COLLATE wrapper on the argument > column-ref's
#    column-default > BINARY. Probe: `CREATE TABLE t(x TEXT COLLATE
#    NOCASE); INSERT VALUES('a'),('B'); SELECT max(x), min(x) FROM t`
#    yields 'B','a' under NOCASE (a < B), not 'a','B' under BINARY
#    ('a'=97 > 'B'=66).
#  * RTRIM column shifts the survivor when trailing spaces matter
#    ('a   ' / 'a' tie under RTRIM; sqlite3 keeps insertion-first).
#  * Explicit `max(x COLLATE NOCASE)` on a plain column applies NOCASE
#    even when the column itself is BINARY.
#  * Numeric (no-collation) MIN/MAX is unaffected — regression-lock.
#  * GROUP BY + max per-group exercises the same compare_kind path.
CREATE TABLE t(x TEXT COLLATE NOCASE); INSERT INTO t VALUES('a'),('B'); SELECT max(x), min(x) FROM t;
CREATE TABLE t(x TEXT COLLATE RTRIM); INSERT INTO t VALUES('a   '),('a'),('b'); SELECT max(x), min(x) FROM t;
SELECT max(x COLLATE NOCASE), min(x COLLATE NOCASE) FROM (SELECT 'a' AS x UNION ALL SELECT 'B');
SELECT max(x), min(x) FROM (VALUES(1.5),(2),(0.5));
CREATE TABLE t(g INT, x TEXT COLLATE NOCASE); INSERT INTO t VALUES(1,'a'),(1,'B'),(2,'C'),(2,'b'); SELECT g, max(x) FROM t GROUP BY g ORDER BY g;

# Read-only PRAGMA defaults (Iter31.W):
#  * sqlite3 freshly-created database defaults — sqlite0 hard-codes
#    matching constants in engine_meta.executePragma. Callers (ORMs,
#    sqlite3 CLI startup probes, GUI tools) routinely query these for
#    feature detection. Setting via `=` is deferred (parser does not
#    yet accept the `=` form for PRAGMA).
PRAGMA encoding;
PRAGMA user_version;
PRAGMA application_id;
PRAGMA page_size;
PRAGMA foreign_keys;
PRAGMA auto_vacuum;
PRAGMA temp_store;
PRAGMA recursive_triggers;
PRAGMA defer_foreign_keys;

# Unknown PRAGMA silent no-op (Iter31.X):
#  * sqlite3 silently ignores PRAGMAs it doesn't recognize (many builds
#    differ in which pragmas exist; tools probe optional ones at setup
#    time). sqlite0 used to error with SyntaxError, blocking legitimate
#    clients. Now returns an empty SELECT result — same exit code 0,
#    no output.
#  * Single-ident arg form (`PRAGMA bogus(passive)`) also silent.
#  * `case_sensitive_like` and similar Iter31.W deferred-set members
#    that have specific sqlite3 behavior (no-row vs single-int) are
#    captured here as the empty-result fall-through (sqlite3 also
#    returns no rows for case_sensitive_like).
PRAGMA bogus_pragma_name;
PRAGMA case_sensitive_like;
PRAGMA something_unknown(passive);

# PRAGMA name = N set form (Iter31.Y):
#  * `PRAGMA user_version` and `PRAGMA application_id` are persisted in
#    the connection (in-memory; file-mode header byte storage deferred).
#    Both read (`PRAGMA name`) and set (`PRAGMA name = N`, `PRAGMA name(N)`).
#  * sqlite3 stores both as i32 — values outside that range silently
#    coerce to 0 (probe `PRAGMA application_id = 4294967295` returns 0).
#    Negative values are valid.
#  * Multi-statement fixture: set then read in one line via `;` separator.
#  * Other PRAGMAs silently accept set forms but discard the value,
#    matching sqlite3's "unknown set is silent" rule.
PRAGMA user_version = 5; PRAGMA user_version;
PRAGMA application_id = 100; PRAGMA application_id;
PRAGMA user_version = -5; PRAGMA user_version;
PRAGMA user_version = 2147483647; PRAGMA user_version;
PRAGMA user_version = 2147483648; PRAGMA user_version;
PRAGMA user_version(7); PRAGMA user_version;
PRAGMA totally_unknown = 99;
