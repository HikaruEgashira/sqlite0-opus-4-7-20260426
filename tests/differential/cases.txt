#    the lenient prefix → REAL → i64 fallback.
SELECT printf('%d', 'inf')
SELECT printf('%d', 'nan')
SELECT printf('%d', 'Inf')
SELECT printf('%d', 'Infinity')
SELECT printf('%d', '-inf')
SELECT printf('%d', 'inf99')
SELECT printf('%d', '9.99e99')
SELECT printf('%d', '1e99')
SELECT printf('%d', '1e-99')
SELECT printf('%d', '.5')
SELECT printf('%i', 'inf')
SELECT printf('%u', 'inf')
SELECT printf('%x', 'inf')
SELECT printf('%o', 'inf')
